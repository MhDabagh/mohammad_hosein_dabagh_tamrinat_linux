# علت: تمام فرآیندهای دیگر از آن ایجاد می‌شوند.

s2:
# پیدا کردن شماره سیگنال SIGTSTP
kill -l | grep SIGTSTP
# نتیجه معمولا 20 است.

s3:
# تفاوت:
# SIGTERM پایان معمولی و قابل مدیریت است.
# SIGKILL پایان اجباری و غیرقابل جلوگیری است.
# زمانی از SIGKILL استفاده می‌شود که فرآیند به SIGTERM پاسخ ندهد.

s4:
# تفاوت دو دستور:
cat file.txt | grep "word"
grep "word" file.txt
# مورد اول شامل استفاده غیرضروری از cat است.

s5:
# فیلتر پیام‌های کرنل که شامل error یا fail باشند
dmesg | grep -Ei "error|fail" > kernel_errors.log

s6:
# دستور tr برای تبدیل یا حذف کاراکترها استفاده می‌شود.
